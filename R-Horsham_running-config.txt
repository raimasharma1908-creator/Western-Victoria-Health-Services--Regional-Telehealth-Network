enable secret cisco123

banner motd ^AUTHORISED ACCESS ONLY - WVHS^

!

interface GigabitEthernet0/0
 description LAN-Horsham-Clinic
 ip address 10.7.2.33 255.255.255.240
 no shutdown

!
! --- WAN Interfaces ---
!

interface Serial0/3/0
 description WAN-to-Ballarat
 ip address 10.7.2.86 255.255.255.252
 no shutdown

!
! --- Static Routes ---
!

ip route 0.0.0.0 0.0.0.0 10.7.2.85

!

!

line con 0
 password cisco123
 login

line vty 0 4
 password cisco123
 login

!
end
